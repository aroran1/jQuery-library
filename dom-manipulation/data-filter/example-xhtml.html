Example HTML

<div class="example" id="example1">
	<ul class="tabs">
		<li class="selected"><a href="#">Show all</a></li>
		<li><a href="#" class="data-group_1">Show group 1</a></li>
		<li><a href="#" class="data-group_2">Show group 2</a></li>
		<li><a href="#" class="data-group_2 data-price_200">Show group 2 with price 200</a></li>
		<li><a href="#" class="data-price_300">Show price 300</a></li>
	</ul>
	<div class="content">
		<p>No group. No price</p>
		<p class="data-group_1 data-price_100">Group 1. Price 100</p>
		<p class="data-group_2 data-price_200">Group 2. Price 200</p>
		<p class="data-price_300">No group. Price 300</p>
		<p class="data-group_1 data-price_300">Group 1 Price 300</p>
		<p class="data-group_2 data-price_300">Group 2 Price 300</p>				
		<p class="data-group_all data-price_all">All groups. All prices</p>
	</div>
</div>

Example JS

$(function () {
	$('.example').dataFilter({
		duration: 400,
		filterElements: '.content p'
	});
});

Parameters

duration
String/Integer - duration of filter animation (default: 0)
filterElements
jQuery selector statement - elements to be filtered inside object (default: 'div')
