	<!-- jQuery code -->

Example JS

$(function () {
	// timer countdown
		var liftoffTime = new Date();
		liftoffTime.setDate(liftoffTime.getDate() + 365);

	$('#countDownBlock').countdown({
		until: liftoffTime,  
		compact: true,
		format: 'HMS',
		layout: '<table summary="Countdown timer">	' +
					'<thead>	' +
						'<tr>	' +
							'<th colspan="2">	Hours</th>	' +
							'<th colspan="2">	Minutes</th>	' +
							'<th>Seconds</th>	' +
						'</tr>	' +
					'</thead>	' +
					'<tbody>	' +
						'<tr>	' +
							'<td>	{hnn}</td>	' +
							'<td class="timeFormat">:</td>' +
							
							'<td>{mnn}</td>' +
							'<td class="timeFormat">:</td>' +
							'<td>{snn}</td>' +
						'</tr>' +
					'</tbody>' +
				'</table>' + 
				'<div class="clockOverlay"></div>',
		onExpiry: liftOff
	});

	function liftOff() {

		var liftoffTime = new Date();

		liftoffTime.setHours(liftoffTime.getHours() + 24);
		$('#countDownBlock').countdown('change', { until: liftoffTime });

	};
});
